#отобразить все файлы каталога в т числе скрытые
 ls -a
#команда -ключ аргумент_1 аргумент_2 … 

#отобразить файлы каталога в виде списка с доп инфо,исключая скрытые
 ls -l

#комбинация флагов -a -l
 ls -la

#drwxr-xr-x@   4 user  staff   128 23 мар 15:31 Applications
#первый столбец говорит о типе файла (-обычный файл, d директория, l ссылка)
#и правах доступа(rwxr-xr-x@, где (rwx) относятся к владельцу данного файла, может делать все с файлом)
#(r-x) относятся к пользователям, которые состоят в той же группе, что и владелец файла(не может записывать),
#далее(r-x) относятся ко всем пользователям(не может записывать),
#которые не состоят в группе владельца и не являются владельцем. 
#Далее идет цифра - количество ссылок на данный элемент (4 ссылки).
#Далее идут имя владельца файла(user) и группа(staff). 
#Затем размер объекта(128), дата последнего изменения(23 мар 15:3) и название самого объекта (Applications).

#о правах, расшифоровка
#"r" означает доступ на чтение данного файла,
#"w" - доступ на изменение (запись) данного файла, 
#"x" - доступ на исполнение (запуск) данного файла
#"-" означает отсутсвие доступа к конкретному действию.

#изменение прав, где 741 это представление в двоичной системе
 chmod 741 testfiie
#7 аналогично 111, т.е. rwx 
#4 аналогично 100, т.е. r--
#1 аналогично 001, т.е. --x

#узнать маску для расчета прав в двоичной системе
 umask
#777 - 022 = 755 //маска равна 022
